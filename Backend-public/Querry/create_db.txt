---Table: sport
CREATE TABLE sport (
    sport_id BIGINT PRIMARY KEY,
    sport_name VARCHAR(255)
);

---Table: course
CREATE TABLE course (
    course_id BIGINT PRIMARY KEY,
    sport_id BIGINT,
    title VARCHAR(255),
    content TEXT,
    created_at TIMESTAMP,
    FOREIGN KEY (sport_id) REFERENCES sport(sport_id)
);

---Table: league
CREATE TABLE league (
    league_id BIGINT PRIMARY KEY,
    sport_id BIGINT,
    league_name VARCHAR(255),
    FOREIGN KEY (sport_id) REFERENCES sport(sport_id)
);

---Table: team
CREATE TABLE team (
    team_id BIGINT PRIMARY KEY,
    league_id BIGINT,
    team_name VARCHAR(255),
    FOREIGN KEY (league_id) REFERENCES league(league_id)
);

---Table: player
CREATE TABLE player (
    player_id BIGINT PRIMARY KEY,
    team_id BIGINT,
    firstname VARCHAR(255),
    lastname VARCHAR(255),
    position VARCHAR(50),
    age INT,
    nationality VARCHAR(100),
    weight FLOAT,
    FOREIGN KEY (team_id) REFERENCES team(team_id)
);

---Table: player_stat
CREATE TABLE player_stat (
    player_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    sport_id BIGINT,
    points INT,
    assists INT,
    saves INT,
    minutes_played INT,
    shots_on_target INT,
    fouls INT,
    matches_played INT,
    goals_per_game_average FLOAT,
    saves_percentage FLOAT,
    goals_against_average FLOAT,
    FOREIGN KEY (player_id) REFERENCES player(player_id),
    FOREIGN KEY (sport_id) REFERENCES sport(sport_id)
);

---Table: user
CREATE TABLE user (
    user_id BIGINT PRIMARY KEY,
    username VARCHAR(255),
    password VARCHAR(255),
    firstname VARCHAR(255),
    lastname VARCHAR(255)
);

---Table: role
CREATE TABLE role (
    role_id BIGINT PRIMARY KEY,
    role_name VARCHAR(255),
    role_description VARCHAR(255)
);

---Table: permission
CREATE TABLE permission (
    permission_id BIGINT PRIMARY KEY,
    name VARCHAR(255),
    description VARCHAR(255)
);

---Table: user_role (Composite Key)
CREATE TABLE user_role (
    user_id BIGINT,
    role_id BIGINT,
    PRIMARY KEY (user_id, role_id),
    FOREIGN KEY (user_id) REFERENCES user(user_id),
    FOREIGN KEY (role_id) REFERENCES role(role_id)
);

---Table: role_permission (Composite Key)
CREATE TABLE role_permission (
    role_id BIGINT,
    permission_id BIGINT,
    PRIMARY KEY (role_id, permission_id),
    FOREIGN KEY (role_id) REFERENCES role(role_id),
    FOREIGN KEY (permission_id) REFERENCES permission(permission_id)
);

---Table: favorite
CREATE TABLE favorite (
    favorite_id BIGINT PRIMARY KEY,
    user_id BIGINT,
    player_id BIGINT,
    FOREIGN KEY (user_id) REFERENCES user(user_id),
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: match
CREATE TABLE match (
    match_id BIGINT PRIMARY KEY,
    home_team_id BIGINT,
    away_team_id BIGINT,
    match_date DATE,
    match_time TIME,
    match_status VARCHAR(255),
    home_score INT,
    away_score INT,
    FOREIGN KEY (home_team_id) REFERENCES team(team_id),
    FOREIGN KEY (away_team_id) REFERENCES team(team_id)
);

---Table: ufc_stats
CREATE TABLE ufc_stats (
    ufc_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    ko_tko_wins INT,
    submission_wins INT,
    decision_wins INT,
    significant_strikes_landed INT,
    significant_strikes_attempted INT,
    takedown_accuracy FLOAT,
    submission_attempted INT,
    average_fight_time FLOAT,
    reach FLOAT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: football_stats
CREATE TABLE football_stats (
    football_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    passing_yards INT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: golf_stats
CREATE TABLE golf_stats (
    golf_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    driving_accuracy FLOAT,
    fairway_accuracy FLOAT,
    greens_in_regulation FLOAT,
    scrambling_percentage FLOAT,
    average_putts FLOAT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: tennis_stats
CREATE TABLE tennis_stats (
    tennis_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    aces INT,
    double_faults INT,
    first_serve_percentage FLOAT,
    first_serve_points_won INT,
    break_points_converted INT,
    matches_won INT,
    sets_won INT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: hockey_stats
CREATE TABLE hockey_stats (
    hockey_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    goals INT,
    assists INT,
    plus_minus INT,
    penalty_minutes INT,
    shots_on_goal INT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: basketball_stats
CREATE TABLE basketball_stats (
    basketball_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    points_per_game FLOAT,
    rebounds_per_game FLOAT,
    assists_per_game FLOAT,
    field_goal_percentage FLOAT,
    free_throw_percentage FLOAT,
    three_point_percentage FLOAT,
    steals INT,
    blocks INT,
    turnovers INT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);

---Table: soccer_stats
CREATE TABLE soccer_stats (
    soccer_stat_id BIGINT PRIMARY KEY,
    player_id BIGINT,
    passes_completed INT,
    pass_completion_percentage FLOAT,
    assists INT,
    goals INT,
    dribbles_completed INT,
    tackles INT,
    yellow_cards INT,
    red_cards INT,
    FOREIGN KEY (player_id) REFERENCES player(player_id)
);
